//做第二题，第一题倒过来输出就行
#include<iostream>
using namespace std;
int main(){
	int i,temp;
	int a[10];
	for(i=0;i<10;i++){
		cin>>a[i];
	}
	for(i=0;i<5;i++){
		temp=a[i];
		a[i]=a[9-i];
		a[9-i]=temp;
	}
	for(i=0;i<10;i++){
		cout<<a[i]<<endl;
	}
	return 0;
}
